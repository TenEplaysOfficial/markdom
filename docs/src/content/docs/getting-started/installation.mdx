import { Tabs, TabItem, Aside } from '@astrojs/starlight/components';

Install Markdom using your preferred package manager:

<Tabs syncKey="constellations">
  <TabItem label="NPM" icon="npm">
    ```bash npm install @tenedev/markdom ```
  </TabItem>
  <TabItem label="Yarn" icon="seti:yarn">
    ```bash yarn add @tenedev/markdom ```
  </TabItem>
</Tabs>

<Aside type="caution" title="Watch out!">
  Requirements Node.js 16 or higher.
</Aside>
